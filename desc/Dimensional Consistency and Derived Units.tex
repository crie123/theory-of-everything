\section{Dimensional Analysis}

\subsection{Energy--Work Relation}

\begin{equation}
E = t \cdot g
\end{equation}

\[
[t] = \text{s}, \quad [g] = \text{J·s}^{-1}
\]

\[
\Rightarrow [E] = \text{J}
\]

This confirms that field allowance $g$ represents a rate of structured work.

\subsection{Work Density}

\begin{equation}
\rho_W = \frac{1}{V}\frac{dW}{dt}
\end{equation}

\[
[dW/dt] = \text{J·s}^{-1}, \quad [V] = \text{m}^3
\]

\[
\Rightarrow [\rho_W] = \text{J·s}^{-1}\text{·m}^{-3}
\]

This quantity governs singularity formation and structural instability.

\subsection{Wave Energy Transport}

\begin{equation}
\frac{dE_{\text{wave}}}{dx} \approx 0
\end{equation}

\[
[E_{\text{wave}}] = \text{J}, \quad [x] = \text{m}
\]

Energy conservation during propagation implies that losses occur only at interaction events.

\subsection{Phase Cancellation}

\begin{equation}
\psi_1 + \psi_2 = 0
\end{equation}

Field amplitude cancellation does not imply zero transported energy;
the dimensionality of $\psi$ depends on normalization and interaction mechanism.

\subsection{Condensation Threshold}

\begin{equation}
E_{\text{local}} \ge E_{\text{cond}}
\end{equation}

\[
[E_{\text{cond}}] = \text{J}
\]

This condition is dimensionally equivalent to known pair-production thresholds
and allows direct comparison with astrophysical energy densities.

\subsection{Gravitational Interpretation}

\begin{equation}
a_{\text{grav}} \sim \nabla g
\end{equation}

\[
[\nabla g] = \text{J·s}^{-1}\text{·m}^{-1}
\]

To recover acceleration units:

\begin{equation}
a_{\text{grav}} \sim \frac{\nabla g}{m}
\end{equation}

\[
\Rightarrow [a] = \text{m·s}^{-2}
\]

This shows that gravitational acceleration arises from gradients
of field allowance normalized by mass.

\subsection{Reversibility Limit}

\begin{equation}
\lim_{\text{coherence} \to 1} E_{\text{matter}} \rightarrow E_{\text{wave}}
\end{equation}

Both quantities retain energy dimension (J), confirming consistency
at the theoretical boundary between matter and wave states.
